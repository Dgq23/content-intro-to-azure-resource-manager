# PowerShell #

<# Login to Azure 
If you have MFA enabled you will need to include "-TenantId <tenant id>" #>
Connect-AzAccount 

<# If you have multiple subscriptions you will need to run
Set-AzContext <SubscriptionName> #>

# Create a new Resource Group #
New-AzResourceGroup  -Name psarmrg  -Location <location>
 
<# Deploy file
If you are deploying from a web accessible file use "-TemplateUri <uri of file>" #>
New-AzResourceGroupDeployment  -Name psblanktemplate  -ResourceGroupName psarmrg  -TemplateFile "<file location>"
 
# CLI #

<# Login to Azure 
If you have MFA enabled you will need to include "--tenant <tenant id>" #>
az login

<# If you have multiple subscriptions you will need to run
az account set --subscription <SubscriptionName> #>

# Create a new Resource Group #
az group create --name cliarmrg --location <location>

<# Deploy file
If you are deploying from a web accessible file use "--template-uri" #>
az deployment group create --name cliblanktemplate --resource-group cliarmrg --template-file "<file location>"
